See It In Action

LevLang Code (.lvl)

// Car Racing Game
game "Car Racing" auto_fps

player {
  movement: wasd_arrows
  speed: 7
}

enemy {
  spawn: random_lane
  speed: rand(3, 6)
  collide: gameover
}

spawn_rate: 2sec

ui {
  "Score: {score}" at topleft
}

start()
